I have a couple of pages that need a set of files, so instead of including one by one in all the pages, I created a function that calls requires them all, keeping the callback function.

require_form = function (callback) {
	require([
		"form",
		"validate",
		"multiselect",
		"form_extend"
		], function(){
			require(["my_validate"], callback);
		}
	);
};

Whenever I need to require the form set of files, I call:

require_form(function(){
	alert('All form files are loaded!');
});
